실제 코드 — 1) 번역 script

JavaScriptscripts/translate-post.mjs
import Anthropic from '@anthropic-ai/sdk';
import fs from 'fs/promises';
import path from 'path';
import matter from 'gray-matter';

const client = new Anthropic({ apiKey: process.env.ANTHROPIC_API_KEY });

const LANGS = {
    ko: { name: 'Korean', region: 'KR' },
    ja: { name: 'Japanese', region: 'JP' },
    zh: { name: 'Simplified Chinese', region: 'CN' },
    es: { name: 'Spanish', region: 'ES' },
    de: { name: 'German', region: 'DE' },
    fr: { name: 'French', region: 'FR' },
    pt: { name: 'Portuguese', region: 'BR' },
    ru: { name: 'Russian', region: 'RU' },
    ar: { name: 'Arabic', region: 'SA' },
    hi: { name: 'Hindi', region: 'IN' },
    th: { name: 'Thai', region: 'TH' },
    id: { name: 'Indonesian', region: 'ID' },
    vi: { name: 'Vietnamese', region: 'VN' },
    tr: { name: 'Turkish', region: 'TR' },
    gb: { name: 'British English', region: 'GB' },
};

const PROMPT = (target, post) => `You are an expert SEO translator
for LuckyPlz, a 16-language decision games site.

Translate this English blog post to natural \${target.name}.

Requirements:
- Cultural localization (idioms, examples, currency)
- Keep brand "LuckyPlz" as-is
- Adapt SEO keywords to \${target.name} search behavior
- Preserve markdown structure exactly
- Keep front-matter keys, translate values
- Tone: same friendliness as English

Original post:
---
\${post}
---

Output the translated markdown only, no explanation.`;

async function translateOne(content, lang){
    const response = await client.messages.create({
        model: 'claude-sonnet-4-5-20250929',
        max_tokens: 8192,
        messages: [{ role: 'user', content: PROMPT(LANGS[lang], content) }],
    });
    return response.content[0].text;
}

async function main(){
    const sourceFile = process.argv[2];           // e.g. content/en/lottery-guide.md
    const slug = path.basename(sourceFile, '.md');
    const source = await fs.readFile(sourceFile, 'utf-8');

    console.log(`Translating \${slug} to 15 languages...`);

    const tasks = Object.keys(LANGS).map(async (lang) => {
        const translated = await translateOne(source, lang);
        const outPath = `content/\${lang}/\${slug}.md`;
        await fs.mkdir(path.dirname(outPath), { recursive: true });
        await fs.writeFile(outPath, translated);
        console.log(`  ✓ \${lang}`);
    });

    await Promise.all(tasks);
    console.log('Done.');
}

main().catch(console.error);

실제 코드 — 2) GitHub Actions 자동 trigger

YAML.github/workflows/translate.yml
name: Auto-translate new posts

on:
  push:
    paths:
      - 'content/en/**.md'
    branches: [main]

jobs:
  translate:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
        with:
          fetch-depth: 2

      - uses: actions/setup-node@v4
        with: { node-version: '20' }

      - run: npm ci

      - name: Find changed files
        id: changes
        run: |
          CHANGED=$(git diff --name-only HEAD~1 HEAD | grep '^content/en/.*\\.md$' || true)
          echo "files=$CHANGED" >> $GITHUB_OUTPUT

      - name: Translate
        if: steps.changes.outputs.files
        env:
          ANTHROPIC_API_KEY: ${{ secrets.ANTHROPIC_API_KEY }}
        run: |
          for f in ${{ steps.changes.outputs.files }}; do
            node scripts/translate-post.mjs "$f"
          done

      - name: Create PR
        uses: peter-evans/create-pull-request@v6
        with:
          branch: auto-translate/${{ github.run_id }}
          title: '🌐 Auto-translate: 15 languages'
          body: 'Translated via Claude API. Review per-language tone before merge.'
          commit-message: 'auto: translate to 15 languages'

실제 코드 — 이미지 생성 + overlay

JavaScriptscripts/generate-pins.mjs
import Replicate from 'replicate';
import sharp from 'sharp';
import fs from 'fs/promises';
import path from 'path';

const replicate = new Replicate({ auth: process.env.REPLICATE_API_TOKEN });

// Pin design variants — 6개의 다른 layout
const VARIANTS = [
    { bg: '#FAF7F0', text: '#2C3E50', layout: 'top' },
    { bg: '#2C3E50', text: '#FAF7F0', layout: 'center' },
    { bg: '#C8924E', text: '#FFFFFF', layout: 'bottom' },
    { bg: '#6B8E7F', text: '#FFFFFF', layout: 'left' },
    { bg: '#FBF1E1', text: '#C8924E', layout: 'right' },
    { bg: '#4A6B7F', text: '#FAF7F0', layout: 'split' },
];

async function generateBase(prompt) {
    const out = await replicate.run(
        "stability-ai/sdxl:39ed52f...",   // version hash
        {
            input: {
                prompt: `\${prompt}, vertical 9:16 social media pin, vibrant, eye-catching`,
                aspect_ratio: '9:16',
                width: 1080, height: 1920,
                num_outputs: 5,
            },
        }
    );
    return out;   // returns array of image URLs
}

async function overlayText(baseUrl, variant, headline, idx, outDir) {
    const baseBuf = await fetch(baseUrl).then(r => r.arrayBuffer());

    // 텍스트 SVG 생성
    const svg = `<svg width="1080" height="1920" xmlns="http://www.w3.org/2000/svg">
        <rect width="1080" height="1920" fill="\${variant.bg}" opacity="0.85"/>
        <text x="540" y="\${variant.layout === 'top' ? 200 : 960}"
              font-family="Inter,sans-serif" font-size="80" font-weight="900"
              fill="\${variant.text}" text-anchor="middle">
            \${headline}
        </text>
        <text x="540" y="\${variant.layout === 'top' ? 280 : 1040}"
              font-family="Inter" font-size="36" fill="\${variant.text}"
              text-anchor="middle" opacity="0.75">
            luckyplz.com
        </text>
    </svg>`;

    await sharp(Buffer.from(baseBuf))
        .composite([{ input: Buffer.from(svg), top: 0, left: 0 }])
        .jpeg({ quality: 90 })
        .toFile(path.join(outDir, `pin-\${idx}.jpg`));
}

async function generateForPost(slug, frontmatter) {
    const outDir = `assets/pins/\${slug}`;
    await fs.mkdir(outDir, { recursive: true });

    // 6 angle 의 prompt 생성
    const angles = [
        `\${frontmatter.topic} tutorial illustration, minimal flat design`,
        `Person playing \${frontmatter.topic}, dynamic action`,
        `\${frontmatter.topic} statistics infographic style`,
        `\${frontmatter.topic} fun cartoon style`,
        `\${frontmatter.topic} retro gaming aesthetic`,
    ];

    let pinIdx = 0;
    for (const prompt of angles) {
        const urls = await generateBase(prompt);
        for (const url of urls) {
            const variant = VARIANTS[pinIdx % VARIANTS.length];
            const headlines = frontmatter.pin_headlines;   // 6개 미리 작성
            await overlayText(url, variant, headlines[pinIdx % 6], pinIdx, outDir);
            pinIdx++;
        }
    }
    console.log(`  ✓ \${pinIdx} pins for \${slug}`);
}

SYSTEM 03

키워드 ranking 모니터링 + Slack 알림

매일 cron 으로 top 200 keyword ranking 조회 → 11-30위 발견 시 즉시 알림 (가장 ROI 높은 boost 기회).

build 시간10-15시간
유지/주0.5시간
productivity3×
월 비용$0 (Search Console API free)
스택Search Console API · Discord webhook · GitHub Actions cron

왜 11-30위 가 핵심인가?

  • 1-10위: 이미 1페이지. 효과 大 but 어렵게 도달.
  • 11-30위: 1-2페이지 사이. 작은 boost (백링크 5-10개, 컨텐츠 보강) 으로 1페이지 진입 가능. 가장 ROI 높음.
  • 30위+: 큰 작업 필요. 우선순위 낮음.

실제 코드

JavaScriptscripts/rank-monitor.mjs
import { google } from 'googleapis';
import fs from 'fs/promises';

const auth = new google.auth.GoogleAuth({
    keyFile: 'service-account.json',
    scopes: ['https://www.googleapis.com/auth/webmasters.readonly'],
});

const searchconsole = google.searchconsole({ version: 'v1', auth });
const SITE = 'sc-domain:luckyplz.com';

async function getRankings() {
    const today = new Date();
    const week_ago = new Date(today.getTime() - 7 * 86400000);

    const res = await searchconsole.searchanalytics.query({
        siteUrl: SITE,
        requestBody: {
            startDate: week_ago.toISOString().split('T')[0],
            endDate: today.toISOString().split('T')[0],
            dimensions: ['query', 'page'],
            rowLimit: 1000,
        },
    });

    return res.data.rows
        .filter(r => r.position >= 11 && r.position <= 30)   // sweet spot
        .filter(r => r.impressions >= 50)                     // 의미 있는 노출
        .sort((a, b) => a.position - b.position);
}

async function loadPrevious() {
    try { return JSON.parse(await fs.readFile('rank-history.json', 'utf-8')); }
    catch { return {}; }
}

async function notify(opportunities) {
    const message = {
        embeds: [{
            title: '🎯 Ranking Opportunities — 11-30위 키워드',
            description: '이 키워드들 boost 시 1페이지 진입 가능. ROI 가장 높음.',
            color: 13144654,   // gold
            fields: opportunities.slice(0, 10).map(o => ({
                name: `\${Math.round(o.position)}위 · "\${o.keys[0]}"`,
                value: `📄 \${o.keys[1]}\\n👁 \${o.impressions} 노출 · CTR \${(o.ctr*100).toFixed(1)}%`,
                inline: false,
            })),
            footer: { text: 'LuckyPlz Rank Monitor' },
            timestamp: new Date().toISOString(),
        }],
    };

    await fetch(process.env.DISCORD_WEBHOOK, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(message),
    });
}

async function main() {
    const current = await getRankings();
    const prev = await loadPrevious();

    // 신규 진입 키워드 (이전엔 30위 밖이었던)
    const newEntries = current.filter(c => {
        const key = `\${c.keys[0]}|\${c.keys[1]}`;
        return !prev[key] || prev[key] > 30;
    });

    if (newEntries.length > 0) {
        await notify(newEntries);
    }

    // history 업데이트
    const next = {};
    for (const c of current) {
        next[`\${c.keys[0]}|\${c.keys[1]}`] = c.position;
    }
    await fs.writeFile('rank-history.json', JSON.stringify(next, null, 2));
}

main().catch(console.error);

핵심 logic

경쟁사가 받은 백링크 = 우리도 받을 수 있는 기회. 같은 사이트가 "wheelofnames" 인용했다면 우리 사이트도 인용 후보.

실제 코드

JavaScriptscripts/competitor-backlinks.mjs
import Anthropic from '@anthropic-ai/sdk';

const COMPETITORS = [
    'wheelofnames.com',
    'random.org',
    'pickerwheel.com',
    'spinthewheel.io',
];

async function getNewBacklinks(domain, since) {
    const r = await fetch(
        `https://apiv2.ahrefs.com/?from=backlinks&target=\${domain}` +
        `&mode=domain&limit=100&order_by=first_seen:desc&output=json`,
        { headers: { 'Authorization': `Bearer \${process.env.AHREFS_TOKEN}` } }
    );
    const data = await r.json();
    return data.refdomains.filter(b => new Date(b.first_seen) > since);
}

const OUTREACH_PROMPT = (link) => `Generate a personalized outreach email
to the editor of \${link.referring_page}.

Context:
- They wrote about \${link.target_url} on their site
- We have a similar/better tool at https://luckyplz.com
- Our angle: 16 languages, 13 game types (broader than competitor)

Email rules:
- Max 100 words
- No marketing speak, conversational
- Acknowledge their article specifically
- Soft suggestion to add LuckyPlz as alternative
- End with low-stakes ask ("Just thought you might like to know")

Output: subject + body.`;

async function generateOutreach(link) {
    const client = new Anthropic({ apiKey: process.env.ANTHROPIC_API_KEY });
    const r = await client.messages.create({
        model: 'claude-sonnet-4-5-20250929',
        max_tokens: 500,
        messages: [{ role: 'user', content: OUTREACH_PROMPT(link) }],
    });
    return r.content[0].text;
}

async function main() {
    const since = new Date(Date.now() - 7 * 86400000);
    const queue = [];

    for (const domain of COMPETITORS) {
        const links = await getNewBacklinks(domain, since);
        for (const l of links.slice(0, 20)) {
            const email = await generateOutreach(l);
            queue.push({
                source: domain,
                link: l.referring_page,
                contact: l.contact_email || 'manual_find_required',
                draft: email,
            });
        }
    }

    // CSV 로 저장 → 본인이 manual review + send
    const csv = queue.map(q =>
        `"\${q.source}","\${q.link}","\${q.contact}","\${q.draft.replace(/"/g, '""')}"`
    ).join('\\n');

    await fs.writeFile(`outreach-\${Date.now()}.csv`, csv);
    console.log(`Generated \${queue.length} outreach drafts`);
}
자동 send 절대 X. spam filter 트리거 + ESP (이메일 service provider) blacklist. 본인이 manual review 한 후 (15-30/일) 개인 메일 클라이언트로 보낼 것.

예상 ROI

  • 주 50 outreach → 평균 3-5% 응답 → 주 2-3 backlink 확보
  • 월 8-12 backlink × 12개월 = 100+ backlink/년
  • 도메인 권위 DR +15-25 (1년 안에)

실제 코드 — Worker

JavaScriptworkers/ab-test.js
// Cloudflare Worker — A/B test middleware
// Runs on every request, assigns user to variant, injects experiment cookie.

const EXPERIMENTS = {
    'ad-placement-v1': {
        variants: ['top-banner', 'sidebar', 'in-content'],
        weights: [33, 33, 34],
        active: true,
        endsAt: '2026-06-01',
    },
    'cta-color-v1': {
        variants: ['cyan', 'gold', 'rose'],
        weights: [50, 25, 25],
        active: true,
        endsAt: '2026-05-15',
    },
};

function pickVariant(userId, exp) {
    // Deterministic hash → 같은 user 는 항상 같은 variant
    const hash = [...userId].reduce((a, c) => a + c.charCodeAt(0), 0);
    const r = hash % 100;
    let cumulative = 0;
    for (let i = 0; i < exp.variants.length; i++) {
        cumulative += exp.weights[i];
        if (r < cumulative) return exp.variants[i];
    }
    return exp.variants[0];
}

export default {
    async fetch(request, env) {
        const url = new URL(request.url);
        let cookies = request.headers.get('Cookie') || '';

        // 사용자 ID — 첫 방문 시 생성, cookie 영구 저장
        let userId = cookies.match(/lp_uid=([^;]+)/)?.[1];
        const isNew = !userId;
        if (isNew) {
            userId = crypto.randomUUID();
        }

        // 활성 실험 모두에 대해 variant 할당
        const assignments = {};
        for (const [name, exp] of Object.entries(EXPERIMENTS)) {
            if (!exp.active) continue;
            if (new Date(exp.endsAt) < new Date()) continue;
            assignments[name] = pickVariant(userId, exp);
        }

        // origin 으로 fetch + variant header 주입
        const response = await fetch(request, {
            headers: { ...request.headers,
                'X-Experiments': JSON.stringify(assignments) },
        });

        // 응답에 cookie + experiment data 주입
        const newResponse = new Response(response.body, response);
        if (isNew) {
            newResponse.headers.append('Set-Cookie',
                `lp_uid=\${userId}; Max-Age=31536000; Path=/; SameSite=Lax; Secure`);
        }
        newResponse.headers.set('X-User-Variants', JSON.stringify(assignments));

        // KV 에 event 기록 (분석용)
        await env.AB_EVENTS.put(`view:\${userId}:\${Date.now()}`,
            JSON.stringify({ url: url.pathname, assignments }),
            { expirationTtl: 86400 * 90 });

        return newResponse;
    },
};

분석 script — 통계 검증

JavaScriptscripts/ab-analyze.mjs
// chi-square test for proportion difference
function chiSquare(success_a, total_a, success_b, total_b) {
    const p_a = success_a / total_a;
    const p_b = success_b / total_b;
    const p_pooled = (success_a + success_b) / (total_a + total_b);
    const se = Math.sqrt(p_pooled * (1 - p_pooled) * (1/total_a + 1/total_b));
    const z = (p_a - p_b) / se;
    const p_value = 2 * (1 - normalCDF(Math.abs(z)));
    return { z, p_value, lift: ((p_b - p_a) / p_a * 100).toFixed(2) };
}

function normalCDF(x) {
    // Abramowitz & Stegun approximation
    const t = 1 / (1 + 0.2316419 * x);
    const d = 0.3989422804 * Math.exp(-x*x/2);
    return 1 - d * t * (0.31938153 + t * (-0.356563782 +
            t * (1.781477937 + t * (-1.821255978 + t * 1.330274429))));
}

async function analyze(experimentName) {
    // KV scan 으로 데이터 모으기 (실제로는 BigQuery/Plausible 권장)
    const events = await loadEvents(experimentName);

    const stats = {};
    for (const e of events) {
        const v = e.variant;
        stats[v] = stats[v] || { views: 0, conversions: 0 };
        stats[v].views++;
        if (e.converted) stats[v].conversions++;
    }

    // baseline 대비 비교
    const variants = Object.keys(stats);
    const baseline = variants[0];
    const results = [];
    for (let i = 1; i < variants.length; i++) {
        const v = variants[i];
        const test = chiSquare(
            stats[baseline].conversions, stats[baseline].views,
            stats[v].conversions, stats[v].views
        );
        results.push({
            variant: v,
            baseline_cvr: (stats[baseline].conversions / stats[baseline].views * 100).toFixed(2),
            variant_cvr: (stats[v].conversions / stats[v].views * 100).toFixed(2),
            lift: test.lift + '%',
            p_value: test.p_value.toFixed(4),
            significant: test.p_value < 0.05,
        });
    }

    console.table(results);

    // p < 0.05 + sample size 충분 (각 variant 1000+) 이면 winner 자동 판정
    const winners = results.filter(r => r.significant && parseFloat(r.lift) > 0);
    if (winners.length > 0) {
        console.log(`\\n🏆 Winner detected: \${winners[0].variant}`);
        console.log(`   → 자동 deploy 가능 (config 업데이트)`);
    }
}
chi-square test · z-score · normal CDF — 박사 통계학 그대로. 일반 마케터가 못 하는 영역. Bayesian A/B test 로 발전 시 더 정밀.
